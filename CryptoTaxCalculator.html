    <script>

      function calculateTax() {

        const buyPrice = document.getElementById("buy-price").value;

        const sellPrice = document.getElementById("sell-price").value;

        const quantity = document.getElementById("quantity").value;

        const taxRate = document.getElementById("tax-rate").value / 100;

        const taxFreeThreshold = document.getElementById("tax-free-threshold").value;


        let capitalGain = (sellPrice - buyPrice) * quantity;


        if (taxFreeThreshold && capitalGain <= taxFreeThreshold) {

          capitalGain = 0;

        }


        const taxAmount = capitalGain * taxRate;


        document.getElementById("tax-amount").innerHTML = `Your tax liability is: $${taxAmount.toFixed(2)}`;

      }

    </script>
